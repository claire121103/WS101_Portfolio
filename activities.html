<!DOCTYPE html>

<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Interactive Paragraph</title>
  <style>
    #demo {
      font-size: 16px;
      color: black;
      background-color: white;
      padding: 10px;
      border: 1px solid black;
      display: block;
    }
  </style>
</head>
<body>
  <h1>Interactive Webpage</h1>
  <p id="demo">
    I am Claire C. Martillano, a 1st year BSIT Student. This is my Activity in creating a simple interactive webpage from scratch using CodePen.io that demonstrates dynamic styling with JavaScript.
  </p>

  <button id="styleButton" onclick="changeStyle()">Change Style</button>
  <button onclick="resetStyle()">Reset Style</button>
  <button onclick="updateText()">Update Text</button>
  <button onclick="toggleVisibility()">Toggle Visibility</button>

  <script>
    function changeStyle() {
      const demo = document.getElementById("demo");
      demo.style.color = "white";
      demo.style.backgroundColor = "black";
      demo.style.fontSize = "20px";
    }

    function resetStyle() {
      const demo = document.getElementById("demo");
      demo.style.color = "black";
      demo.style.backgroundColor = "white";
      demo.style.fontSize = "16px";
    }

    function updateText() {
      const demo = document.getElementById("demo");
      demo.innerHTML = "The text has been updated!";
    }

    function toggleVisibility() {
      const demo = document.getElementById("demo");
      if (demo.style.display === "none") {
        demo.style.display = "block";
      } else {
        demo.style.display = "none";
      }
    }
  </script>

</body>
</html>

JavaScript

let colors = ["red", "blue", "yellow"];
let currentIndex = 0;

function changeStyle() {
  let para = document.getElementById("demo");
  para.style.fontSize = "20px";
  para.style.color = "white";
  para.style.backgroundColor = colors[currentIndex];

  currentIndex = (currentIndex + 1) % colors.length;
}

function resetStyle() {
  let para = document.getElementById("demo");
  para.style.fontSize = "16px";
  para.style.color = "black";
  para.style.backgroundColor = "white";
}

function updateText() {
  document.getElementById("demo").innerText = "The text has been updated!";
}

function toggleVisibility() {
  let para = document.getElementById("demo");
  if (para.style.display === "none") {
    para.style.display = "block";
  } else {
    para.style.display = "none";
  }
}

document.getElementById("demo").addEventListener("click", function () {
  alert("You clicked the paragraph!");
});
